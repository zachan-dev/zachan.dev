public void TestSetLinks()
{    
  TSTestFactory tsFact;
  TestSet aTestSet;
  List tsSets, linksList;
  ILinkable linkableTestSet;

  TestSetFactory tSetFact = (TestSetFactory)tdConnection.TestSetFactory;
  String tempStr = String.Empty;

  tsSets = tSetFact.NewList("");
  aTestSet = tsSets[2];
  tempStr = String.Format("{0}  {1}", aTestSet.ID, aTestSet.Name);
  Debug.Print(tempStr);
  
  linkableTestSet = (ILinkable)aTestSet;
  LinkFactory linkF = linkableTestSet.BugLinkFactory;

  linksList = linkF.NewList("");
  tempStr = linksList.Count.ToString();
  Debug.Print(tempStr);
}
