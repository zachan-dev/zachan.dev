Get the child release folders of a specified release folder
Public Sub FindReleaseFoldersByParentId()

' Get the child release folders of a specified release folder

    Dim folderFactory As ReleaseFolderFactory
    Dim RelFolderFilter As TDFilter
    Dim listOfFolders As List
    Dim tempFolder As ReleaseFolder

    ' Get ReleaseFolderFactory from TDConnection.
    Set folderFactory = tdc.ReleaseFolderFactory
    Set RelFolderFilter = folderFactory.Filter

    'Get the list of the ReleaseFolders under the
    ' specified node.
    Dim ParentId As Long
    ParentId = 102
    RelFolderFilter.Filter("RF_PARENT_ID") = ParentId
    Set listOfFolders = RelFolderFilter.NewList


    For Each tempFolder In listOfFolders
        Debug.Print tempFolder.Name
    Next


    Set folderFactory = Nothing
    Set RelFolderFilter = Nothing
    Set listOfFolders = Nothing
    Set tempFolder = Nothing

End Sub
