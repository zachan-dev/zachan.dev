Public Sub UpdateActions()

' Update actions

    Dim allActions As CustomizationActions
    Dim action As CustomizationAction
    
    Set CustObj = tdc.Customization
    Set allActions = CustObj.Actions
    
    Set action = allActions.action("ac_modify_run")
    Dim a As Boolean
    a = action.AuditAction
    MsgBox "Audit event ac_modify_run as " & Str(a)
    action.AuditAction = Not a
    a = action.AuditAction
    MsgBox "Now audit event ac_modify_run " & Str(a)

    Set action = allActions.action("ac_delete_bug")
    a = action.AuditAction
    MsgBox "Audit event ac_delete_bug " & Str(a)
    action.AuditAction = Not a
    a = action.AuditAction
    MsgBox "Now audit event ac_delete_bug " & Str(a)
    
    CustObj.Commit
    
End Sub
