Get requirement and test coverage fields
Private Sub Get_Coverage_Fields()

' Get requirement and test coverage fields

Dim oList As List
Dim FieldProp As FieldProperty
Dim oField As TDField

Set oList = tdc.Fields("REQ_COVERAGE_TYPE")

Dim Item
For Each Item In oList
    Set oField = Item
    Set FieldProp = oField.Property
    Debug.Print FieldProp.DBColumnName & " " & FieldProp.DBTableName
    Debug.Print FieldProp.DBColumnName & " " & FieldProp.UserLabel
Next


Set oList = tdc.Fields("TEST_COVERAGE_TYPE")

For Each Item In oList
    Set oField = Item
    Set FieldProp = oField.Property
    'Debug.Print fieldprop.DBColumnName & " " & fieldprop.UserLabel
    Debug.Print FieldProp.DBColumnName & " " & FieldProp.DBTableName
Next

End Sub
