Private Sub CutPasteTestSet()

' Move test set
    
    Dim testSetF As TestSetFactory
    Dim testSet1 As TestSet
    
    Dim TestSetID%
    TestSetID = 64
    
    Set testSetF = tdc.TestSetFactory
    Set testSet1 = testSetF.Item(TestSetID)
    testSet1.TestSetFolder = 5
    testSet1.Post
    
End Sub
