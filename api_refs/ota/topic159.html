public List FilterDefects()
{
  BugFactory bugF;
  TDFilter bugFFilter;
  List bugL;

  String tempStr = String.Empty;

  //Setting filters on multiple fields
  bugF = (BugFactory)tdConnection.BugFactory;
  bugFFilter = bugF.Filter;
  bugFFilter["BG_RESPONSIBLE"] = "alex_alm Or cecil_alm";
  bugFFilter["BG_DETECTION_DATE"] = "[PreviousWeek]";
  bugFFilter["BG_STATUS"] = "New  Or  Open  Or  Rejected  Or  Reopen ";

  bugL = bugFFilter.NewList();
  foreach (Bug thisBug in bugL)
  {
    tempStr = String.Format("{0}  {1}", thisBug.ID, thisBug.Summary);
    Debug.Print(tempStr);
  }
  return bugL;
}
