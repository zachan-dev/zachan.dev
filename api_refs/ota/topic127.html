Public Sub SetGroupPermissions()
' Set group permissions

    Dim custom As Customization
    Set custom = tdc.Customization
    Dim customGroups As CustomizationUsersGroups
    Set customGroups = custom.UsersGroups
    Dim oPermissions As CustomizationPermissions
    Set oPermissions = custom.Permissions
    
    Dim Group As CustomizationUsersGroup
    Set Group = customGroups.Group("NetworkSupport")
    
    oPermissions.CanAddItem("HOSTS", Group) = True
    oPermissions.CanAddItem("HOST_GROUP", Group) = True
    oPermissions.CanAddItem("HOST_IN_GROUP", Group) = True
    oPermissions.CanModifyItem("HOSTS", Group) = True
    oPermissions.CanModifyItem("HOST_GROUP", Group) = True
    oPermissions.CanModifyItem("HOST_IN_GROUP", Group) = True
    
    Group.Updated = True
    custom.Commit

End Sub
