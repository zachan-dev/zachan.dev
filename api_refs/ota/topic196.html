Build a summary graph of Tests
Public Function BuildSummaryGraphOfTests() As Graph

' Build a summary graph of Tests

    Dim testF As TestFactory
    Dim graph1 As Graph
    Dim aFilter As TDFilter
    Dim fRDate As String
    
    Set testF = tdc.TestFactory
    Set aFilter = testF.Filter
    Set graph1 = _
     testF.BuildSummaryGraph("TS_STATUS", "TS_STATUS", "", 0, aFilter, False, False)
            
    Debug.Print graph1.ColCount
    Debug.Print graph1.GraphTotal
    Debug.Print graph1.MaxValue
    Debug.Print graph1.RowCount
    
    Set BuildTrendGraphOfTests = graph1
Exit Function
FUNC_ERR:
     Set BuildTrendGraphOfTests = Nothing
End Function
