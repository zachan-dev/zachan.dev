Public Sub GetAuditLogs()

' Get all audit logs.
 
    Dim TstFilterList As TDAPIOLELib.List
    Dim AuditRecF As AuditRecordFactory
    
'tdc is a TDConnection object. The user is authenticated and
' connected to a project before this routine is run.
    Set AuditRecF = tdc.AuditRecordFactory
    
    Dim testFilter As TDAPIOLELib.TDFilter
' Get list of audit records on changes in defect status.
    Set testFilter = AuditRecF.Filter
    testFilter.Filter("AU_ENTITY_TYPE") = "BUG"
    testFilter.Filter("AP_FIELD_NAME") = "BG_STATUS"
    Set TstFilterList = testFilter.NewList
                     '
    Dim actionID As Long, i&, j&
    Dim msg As String, oldVal$, newVal$
    Dim oldDate As Date, newDate As Date
    Dim AuditRec As AuditRecord
    Dim AuditPropF As AuditPropertyFactory
    Dim propList As TDAPIOLELib.List
' Get the audit records.
    For i = 1 To TstFilterList.Count
        Set AuditRec = TstFilterList.Item(i)
        With AuditRec
            actionID = .actionID
            msg = "SessionID: " & .sessionID & vbCrLf
            msg = msg & "Action: " & .action & vbCrLf
            msg = msg & "Entity Type: " & .entityType & " (" & .entityID & ")" & vbCrLf
            msg = msg & "Description: """ & .Description & """"
        End With
        MsgBox msg
' Get the audit property records.
        Set AuditPropF = AuditRec.AuditPropertyFactory
        Set propList = AuditPropF.NewList("")
        If propList.Count = 0 Then
            MsgBox "No properties for action " & Str(actionID)
        End If
        
        For j = 1 To propList.Count
            Dim ap As AuditProperty
            Set ap = propList.Item(j)
            With ap
                msg = "Action ID: " & Str(actionID) & vbCrLf
                msg = msg & "Field: " & .TableName & "." & .FieldName & vbCrLf
                msg = msg & "FieldName: " & .FieldName & vbCrLf
               
                Dim xx
                xx = ap.oldValue
            End With
            If IsDate(xx) Then
                oldDate = xx: oldVal = ""
            Else
                oldVal = xx: oldDate = Now
            End If
            Dim yy
            yy = ap.newValue
            If IsDate(yy) Then
                newDate = yy: newVal = ""
            Else
                newVal = yy: newDate = Now
            End If
            msg = msg & "Old Value = """ & oldVal & """, New Value = """ & newVal & """" & vbCrLf
            msg = msg & "OldDate = " & oldDate & ", NewDate = " & newDate
            MsgBox msg
        
        Next j
    Next i
    
End Sub
