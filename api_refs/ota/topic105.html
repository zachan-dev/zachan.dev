Get CustomizationMailConditions
Public Sub GetRules()

' Get CustomizationMailConditions

    Dim custMailCondsMgr As CustomizationMailConditions
    Dim custMailCond As CustomizationMailCondition
    Dim conList As List
    'tdc is the global TDConnection object.
    Set custMailCondsMgr = tdc.Customization.MailConditions
    Set conList = custMailCondsMgr.Conditions
    
    For Each custMailCond In conList
        With custMailCond
            Debug.Print "Name = "; .name
            Debug.Print "ConditionText = "; .ConditionText
            Debug.Print "ConditionType = "; .ConditionType
            Debug.Print "---------------------------------"
        End With
    Next
    
End Sub
