Copy and paste a requirement
Private Sub CopyPasteReq(IDSFilter, CopyMode, Path, TargetID, PasteMode, OrderID)

' Copy a requirement

    Dim iscp As ISupportCopyPaste
    Dim clipboard As String
    
    Set iscp = tdc.ReqFactory
    clipboard = iscp.CopyToClipBoard(IDSFilter, CopyMode, Path)
    iscp.PasteFromClipBoard clipboard, TargetID, PasteMode, OrderID
    
    Set iscp = Nothing
End Sub
